````JavaScript
//定义对象接口类型
interface List {
    id:number,
    name:string
}
interface Result{
    data:List[]
}
function render(result:Result){
    result.data.forEach((value)=>{
        console.log(value.id,value.name))
    })
}
let result = {
    data: [
        {id: 1, name: 'A', sex: 'male'},
        {id: 2, name: 'B', age: 10}
    ]
}
render(result)
````

## 函数类型接口
> 使用变量定义一个函数
````JavaScript
//变量定义函数
let add: Add = (a: number, b: number) => a + b
//接口定义函数 直接定义函数的参数类型
interface Add{
    (x:number,y:number):number
}
//使用类型别名
type Add = (x: number, y: number) => number
let add:Add =(a:number,b:number)=>a+b
````
